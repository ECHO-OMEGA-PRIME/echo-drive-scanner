```bash
# Clone
git clone https://github.com/ECHO-OMEGA-PRIME/intelligent-drive-scanner.git
cd intelligent-drive-scanner

# Install dependencies
npm install

# Install Python dependencies
pip install -r requirements.txt

# Run
python main.py
```
